% Headings

\define[2]\StoryTitle{%
	\framed[
    frame=off,
    location=hanging,
    offset=overlay,
    top=\vss,
    bottom=,
    width=broad,
    height=10\lineheight,
    align=middle,
  ]{%
% 		\doiftextelse{\structureuservariable{subtitle}}{%
%       \blank[5*line]%
% 		}{% 
%      \blank[7*line]%
%     }%
		\doiftext{\structureuservariable{author}}{%
		\tf
      \structureuservariable{author}\\%
    }%
		\ss\bfd#2%
		\doiftext{\structureuservariable{subtitle}}{%
			\\\bfa\structureuservariable{subtitle}%
		}
	}
}

\definepagebreak
	[chapterpagebreak]
	[yes,header,footer,right]

\setuphead[
  chapter] [
  header=empty,
  page=chapterpagebreak,
  alternative=command,
  command=\StoryTitle,
  number=no,
  style=\bf
]
